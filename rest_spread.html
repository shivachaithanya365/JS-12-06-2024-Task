Iterate object by using Rest operator


        let obj = {
            state: "Telangana",
            capital: "Hyderabad",
            population: 35000000,
            language: "Telugu"
        };

        function object({ ...rest }) {
            for (let key in rest) {
                console.log(`${key}: ${rest[key]}`);
            }
        }

        object(obj);
